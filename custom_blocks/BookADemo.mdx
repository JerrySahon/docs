export const BookDemoButton = ({
  href = "https://calendly.com/all_pearch/hrachik-and-jerry-15min",
  label = "Book a demo",
  sublabel = "15 minutes • Calendly",
  size = "lg",
  icon = true
}) => {
  const sizes = {
    sm: "text-sm px-4 py-2.5",
    md: "text-base px-5 py-3",
    lg: "text-lg px-6 py-4"
  }[size] || "text-lg px-6 py-4";

  // Определяем, активна ли тёмная тема
  const isDark =
    typeof document !== "undefined" &&
    document.documentElement.classList.contains("dark");

  const bgColor = isDark ? "#ffffff" : "#2a2a2a";
  const textColor = isDark ? "#2a2a2a" : "#ffffff";
  const hoverColor = isDark ? "#f5f5f5" : "#000000";

  return (
    <div className="flex justify-center w-full">
      <a
        href={href}
        target="_blank"
        rel="noopener noreferrer"
        className={[
          "flex items-center gap-3 rounded-2xl shadow-lg transition-all duration-200",
          "w-[100%] justify-center",
          sizes
        ].join(" ")}
        style={{
          backgroundColor: bgColor,
          color: textColor
        }}
        onMouseEnter={(e) =>
          (e.currentTarget.style.backgroundColor = hoverColor)
        }
        onMouseLeave={(e) =>
          (e.currentTarget.style.backgroundColor = bgColor)
        }
      >
        {icon && (
          <i
            className="fa-solid fa-calendar text-xl"
            style={{ color: textColor }}
          />
        )}
        <span className="leading-tight text-center">
          <span className="font-semibold">{label}</span>
          {sublabel && (
            <span
              className="block opacity-80 text-xs font-normal"
              style={{ color: textColor }}
            >
              {sublabel}
            </span>
          )}
        </span>
        <i
          className="fa-solid fa-arrow-right text-lg opacity-80"
          style={{ color: textColor }}
        />
      </a>
    </div>
  );
};
